#include <tunables/global>

profile immich-public-proxy flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>
  #include <abstractions/nameservice>

  # Allow reading and executing the entrypoint script at root
  /run.sh rix,
  
  # Allow common shells used by scripts
  /bin/bash rix,
  /bin/sh rix,
  # ---------------

  # Existing network and data rules
  network inet stream,
  network inet6 stream,
  /data/options.json r,
  /data/ r,
  /data/** rwk,

  # Allow other system binaries
  /usr/bin/** rix,
  /bin/** rix,
  /etc/s6/** rix,
  /run/s6/** rix,
  
  capability setgid,
  capability setuid,
  capability chown,

  signal (send) set=(kill,term,int,hup,cont),
}
